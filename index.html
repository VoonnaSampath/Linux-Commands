grep – Pattern Searching Power Tool

Command Description VLSI Use Case
grep "Error" file.log Search for a word in a file Find errors in synthesis or simulation logs
grep -i "warning" *.rpt Case-insensitive search Capture all warnings in reports
grep -r "module" . Recursive search Find Verilog modules in a project
grep -n "clk" design.v Show line numbers Debug clock-related logic
grep -w "reset" *.v Match whole word only Avoid false signal matches
grep -c "Error" *.log Count matching lines Count total errors per log
grep -v "INFO" run.log Exclude matching lines Hide non-critical messages
grep -E "WNS|TNS" timing.rpt Extended regex search Extract key timing metrics 
# Find all errors and warnings together
grep -E "Error|Warning" synthesis.log

# Search clock related messages ignoring case
grep -i "clk" run.log

# Search pattern in multiple file types
grep "setup" *.rpt *.log

# Show 3 lines before and after a match
grep -n -C 3 "Error" synthesis.log

sed – Stream Editor for Automation

Command Description VLSI Use Case
sed 's/old/new/' file.v Replace first occurrence Rename signals or modules
sed 's/old/new/g' file.v Replace globally Bulk refactoring
sed -n '10,20p' report.rpt Print line range Inspect specific report sections
sed '/^$/d' file.txt Delete empty lines Clean EDA reports
sed '1d' file.log Delete first line Remove headers from reports
sed '$d' file.log Delete last line Remove summary footers
sed 's/#.*//' script.tcl Remove comments Clean TCL scripts 
# Remove blank lines from a report
sed '/^$/d' timing.rpt

# Replace tabs with spaces
sed 's/	/ /g' report.txt

# Comment out lines containing debug
sed '/debug/s/^/#/' script.tcl

# Print only non-comment lines
sed '/^#/d' constraints.sdc

awk – Report Analysis Tool

Command Description VLSI Use Case
awk '{print $1}' file.txt Print first column Extract cell names or paths
awk '{print $1, $3}' report.txt Print selected columns Analyze timing values
awk '{sum+=$2} END {print sum}' data.txt Sum a column Total area or power values
awk '$3 < 0 {print $0}' timing.rpt Conditional printing Find negative slack paths
awk 'NR==1{print}' file.txt Print first line Check report headers
awk 'END{print NR}' file.txt Count lines Count report entries
awk '{max=($2>max)?$2:max} END{print max}' data.txt Find maximum value Identify worst timing path 
# Print paths with slack less than 0
awk '$3 < 0 {print $0}' timing.rpt

# Print column names (first row)
awk 'NR==1 {print}' report.txt

# Print second column only
awk '{print $2}' data.txt

# Print line number with content
awk '{print NR, $0}' file.txt

5 Beginner Shell Scripts That Impress on LinkedIn

1. Error Checker Script

#!/bin/bash
LOG_DIR=logs
OUT_FILE=errors.txt

grep -i "error" $LOG_DIR/*.log > $OUT_FILE
echo "----------------------" >> $OUT_FILE
echo "Total Errors:" >> $OUT_FILE
grep -ic "error" $LOG_DIR/*.log >> $OUT_FILE
echo "Error report generated"

2. Daily Project Backup

#!/bin/bash
PROJECT_DIR=design
DATE=$(date +%Y%m%d)
BACKUP_NAME=backup_$DATE.tar.gz

tar -czf $BACKUP_NAME $PROJECT_DIR
echo "Backup created: $BACKUP_NAME"
ls -lh $BACKUP_NAME

3. Count Verilog Modules

#!/bin/bash
VERILOG_FILES=$(find . -name "*.v")
MODULE_COUNT=$(echo "$VERILOG_FILES" | xargs grep "module" | wc -l)

echo "Total Verilog files:"
echo "$VERILOG_FILES" | wc -l
echo "Total modules:" $MODULE_COUNT

4. Largest Log Files Finder

#!/bin/bash
LOG_DIR=logs

echo "Top 5 largest log files:"
ls -lhS $LOG_DIR/*.log | head -5
echo "------------------------"
du -sh $LOG_DIR/*.log

5. Timing Slack Extractor

#!/bin/bash
INPUT_FILE=timing.rpt
OUTPUT_FILE=negative_slack.txt

awk '$3 < 0 {print $1, $3}' $INPUT_FILE > $OUTPUT_FILE
echo "Negative slack paths saved"
echo "Count:" $(wc -l < $OUTPUT_FILE)
head -5 $OUTPUT_FILE
